Qwirkle Scores

import matplotlib.pyplot as plt import numpy as np import pandas as pd from pyodide.http import open_url url = "https://raw.githubusercontent.com/christyheaton/webmapping_with_pyscript/main/Data/2021-09_qwirkle_scores.csv" scores = pd.read_csv(open_url(url)) scores = scores.set_index("turn") sorted_cumsum = scores.sort_index().cumsum() fig, ax = plt.subplots() plt.xticks(np.arange(20)) ax.set_xlabel("Turn") ax.set_ylabel("Score") ax.plot(sorted_cumsum["b"], label = "b") ax.plot(sorted_cumsum["c"], label = "c") ax.plot(sorted_cumsum["t"], label = "t") ax.legend() display(fig)
