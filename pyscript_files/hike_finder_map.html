packages = ["folium", "pandas", "geopandas", "matplotlib", "mapclassify"] import folium import pandas as pd import geopandas as gpd from pyodide.http import open_url # load and read data url = "https://raw.githubusercontent.com/christyheaton/webmapping_with_pyscript/main/Data/2021-04-25_wta_hike_data.csv" hike_data = pd.read_csv(open_url(url)) # clean data hike_data = hike_data.dropna(subset=['LATITUDE', 'LONGITUDE', 'DISTANCE']) hike_data.at[937,'LATITUDE'] = 48.3093 hike_data.at[937,'LONGITUDE'] = -120.6565 hike_data.at[2412,'LONGITUDE'] = -120.4 # filter data hike_data = hike_data[(hike_data["DISTANCE"] >= 8) & (hike_data["DISTANCE"] <= 20) & (hike_data["DIST_TYPE"] == "roundtrip")].reset_index(drop=True) hike_data = hike_data[hike_data["RATING"] >= 4].reset_index(drop=True) # add geometry hikes_gdf = gpd.GeoDataFrame(hike_data, geometry=gpd.points_from_xy(hike_data.LONGITUDE, hike_data.LATITUDE)) hikes_gdf = hikes_gdf.set_crs("epsg:4326") # create sample sample_hikes = hikes_gdf.sample(n=5) # create interactive map m = sample_hikes.explore(marker_kwds=dict(radius=6)) display(m)
